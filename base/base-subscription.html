    # 观察者与发布订阅区别

    # 观察者模式

    所谓观察者模式，其实就是为了实现松耦合

    观察者模式实现了一对多的依赖关系，当一个对象的某个状态发生变化的时候，所有依赖它的对象都将收到通知，并自动更新。观察者模式属于行为模式，行为型模式关注的是对象之间的通讯；观察者模式就是观察者和被观察者之间的通讯。

    例：
    假设公司一群程序员正在敲代码，这时候前台小姐姐不小心按到了总闸的开关，这时前台小姐姐通知所有程序员赶快保存代码。一个前台小姐姐对应一群程序员，这就是开头说的“一对多”的依赖关系,观察者模式（前台小姐姐是被观察者，所有程序员是观察者）。

    # 代码实现

    //抽象目标
class Subject {
    constructor(name){
        this.name = name;
        this.subs = [];
        this.state = '良好';
    }
    //增加观察者方法
    subscribe(observe){
        this.subs.push(observe);
    }
    //通知观察者方法
    setState(newState){
        this.state = newState;
        this.subs.forEach( c => {
            c.update(newState)
        })
    }
}

// 观察者类
class Observe {
    constructor(name){
        this.name = name;
    }
    // 自身更新状态的方法
    update(data){
        console.log(`${this.name}听说：${data}，赶紧保存了代码`);
    }
}

    # 发布订阅

    其实 24 种基本的设计模式中并没有发布订阅模式，它是只观察者模式的一个别称。但是经过时间的沉淀，似乎它已经强大了起来，已经独立观察者模式，成为另外一种不同的设计模式。大概很多人都和我一样，觉得发布订阅模式里的Publisher，就是观察者模式里的Subject，而Subscriber，就是Observer。Publisher变化时，就主动去通知Subscriber。

    其实并不是。在发布订阅模式里，发布者，并不会直接通知订阅者，换句话说，发布者和订阅者，彼此互不相识。

    如果互不相识的话那它们怎么交流呢？
    答案是，通过第三者，也就是在消息队列里面，我们常说的经纪人Broker。

    • 发布者只需告诉Broker，我要发的消息，topic是AAA；

    • 订阅者只需告诉Broker，我要订阅topic是AAA的消息；

    例：
    你在微博上关注了 A，同时其他很多人也关注了 A，那么当 A 发布动态的时候，微博就会为你们推送这条动态。A 就是发布者，你是订阅者，微博就是调度中心，你和 A 是没有直接的消息往来的，全是通过微博来协调的。

    # 代码实现

    const eventEmitter = {
    // 创建缓存列表
    list: {},
    // 订阅
    on: function(event, fn) {
        let _this = this;
        // 如果缓存列表存在fn就添加到相应缓存列表内，无就初始化一个缓存列表并添加
        (_this.list[event] || (_this.list[event] = [])).push(fn);
        return _this;
    },
    // 发布
    emit: function() {
        let _this = this;
        // 获取event名，
        let event = [].shift.call(arguments);
        // 获取缓存列表
        let fns = [..._this.list[event]];
        // 如果缓存列表无直接返回
        if(!fns || fns.length === 0){
            return false;
        }
        // 如果缓存列表有就依次执行
        fns.forEach(fn => {
            fn.apply(_this, arguments);
        })
        return _this;
    },
    // 取消发布
    off: function(event, fn){
        let _this = this;
        // 获取缓存列表中的fns
        let fns = _this.list[event];
        // 如果没有返回false
        if(!fns) return false;
        // 有没有传fn函数名
        if(!fn){
            // 没有的话缓存列表就清零
            fns && (fns.length = 0);
        }else{
            // 有的话就找出一样的函数删除掉
            let cb;
            for(let i = 0; i < fns.length; i++){
                cb = fns[i];
                if(cb === fn || cb.fn === fn){
                    fns.splice(i, 1);
                    break;
                }
            }
        }
        return _this;
    }
};

    # 输出结果

    function f1(content){
    console.log(`用户1订阅了：${content}`)
}

function f2(content) {
    console.log(`用户2订阅了：${content}`)
}

function f3(content) {
    console.log(`用户3订阅了：${content}`)
}

// 订阅
eventEmitter.on('user1', f1);
eventEmitter.on('user1', f2);

// 取消订阅
eventEmitter.off('user1', f1);
// 订阅
eventEmitter.on('user2', f3);

// 用户2订阅了：js-发布订阅模式
eventEmitter.emit('user1', 'js-发布订阅模式');
// 用户3订阅了：js-观察者模式
eventEmitter.emit('user2', 'js-观察者模式');
